less-is-more implementation

Adicionar o código abaixo ao arquivo swfobject.js "minified". 

// less-is-more implementation 
swfobject._embedSWF = swfobject.embedSWF;
swfobject.embedSWF = null;
swfobject.embedSWF = function(swfUrlStr, replaceElemIdStr, widthStr, heightStr, swfVersionStr, xiSwfUrlStr, flashvarsObj, parObj, attObj){
	// name = id
	if(typeof(attObj) == 'undefined') attObj = {};
	attObj.name = replaceElemIdStr;
	// default parameters
	if(typeof(parObj) == 'undefined') parObj = {};
	if(!parObj.allowscriptaccess) parObj.allowscriptaccess = 'always';
	if(!parObj.quality) parObj.quality = 'best';
	if(!parObj.menu) parObj.menu = false;
	swfobject._embedSWF(swfUrlStr, replaceElemIdStr, widthStr, heightStr, swfVersionStr, xiSwfUrlStr, flashvarsObj, parObj, attObj);
};
